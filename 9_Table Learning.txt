        function openTable(num) {
            let number = num || document.getElementById("tableInput").value;

            if (number < 2 || number > 50 || number === "") {
                alert("Please select a number between 2 and 50.");
                return;
            }

            let newTab = window.open("", "_blank");

            let tableHTML = `
                <html>
                <head>
                    <title>Table of ${number}</title>
                    <style>
                        body {
                            font-family: Arial;
                            background-color: #fff;
                            text-align: center;
                            padding: 20px;
                        }
                        h2 {

                        }
                        p {
                            font-weight: bold;
                        }
                    </style>
                </head>
                <body>
                    <h2>Multiplication Table of ${number}</h2>
                    <p>Range: 2 to 50</p>
            `;

            for (let i = 1; i <= 10; i++) {
                tableHTML += `<p>${number} Ã— ${i} = ${number * i}</p>`;
            }

            tableHTML += `
                </body>
                </html>
            `;

            newTab.document.write(tableHTML);
            newTab.document.close();
        }
    </script>
